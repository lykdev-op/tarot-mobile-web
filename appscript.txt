function doPost(e) {
  try {
    if (!e || !e.postData || !e.postData.contents) {
      throw new Error('No request body received.');
    }

    var payload = JSON.parse(e.postData.contents);
    var result = getTarotReading(payload);

    return jsonResponse({
      success: true,
      html: result.html
    });

  } catch (error) {
    return jsonResponse({
      success: false,
      error: error.message
    });
  }
}

function doGet(e) {
  return jsonResponse({
    success: true,
    message: 'CardReadBackend is running.'
  });
}

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function getTarotReading(payload) {
  validatePayload(payload);

  var apiKey = PropertiesService
    .getScriptProperties()
    .getProperty('OPENAI_API_KEY');

  if (!apiKey) {
    throw new Error('OPENAI_API_KEY is not set in Script Properties.');
  }

  var prompt = payload.lang === 'zh-TW'
    ? buildPromptZh(payload)
    : buildPromptEn(payload);

  var response = UrlFetchApp.fetch('https://api.openai.com/v1/chat/completions', {
    method: 'post',
    headers: {
      Authorization: 'Bearer ' + apiKey,
      'Content-Type': 'application/json'
    },
    payload: JSON.stringify({
      model: 'gpt-4o',
      max_tokens: 2048,
      messages: [
        {
          role: 'user',
          content: prompt
        }
      ]
    }),
    muteHttpExceptions: true
  });

  var statusCode = response.getResponseCode();
  var responseText = response.getContentText();

  if (statusCode !== 200) {
    throw new Error('OpenAI API returned status ' + statusCode + ': ' + responseText);
  }

  var json = JSON.parse(responseText);

  if (!json.choices || !json.choices[0] || !json.choices[0].message) {
    throw new Error('Unexpected OpenAI response format.');
  }

  var rawText = json.choices[0].message.content;

  return {
    html: formatReadingAsHtml(rawText),
    text: rawText
  };
}

function validatePayload(payload) {
  if (!payload) {
    throw new Error('Missing payload.');
  }

  if (!payload.category) {
    throw new Error('Missing category.');
  }

  if (!payload.formation || !payload.formation.name) {
    throw new Error('Missing formation.');
  }

  if (!payload.cards || !Array.isArray(payload.cards) || payload.cards.length === 0) {
    throw new Error('Missing cards.');
  }
}

// ── ENGLISH PROMPT ────────────────────────────────────────────────────────────
// Tone: Reddit roast mode — sarcastic, sharp, but genuine tarot wisdom underneath.

function buildPromptEn(payload) {
  var cardLines = payload.cards.map(function(entry, i) {
    var positionName = 'Position ' + (i + 1);

    if (
      payload.formation &&
      payload.formation.positions &&
      payload.formation.positions[i]
    ) {
      positionName = payload.formation.positions[i];
    }

    var cardName = entry.card && (entry.card.displayName || entry.card.name)
      ? (entry.card.displayName || entry.card.name)
      : 'Unknown Card';

    return (
      '  Position ' +
      (i + 1) +
      ' — ' +
      positionName +
      ': ' +
      cardName +
      (entry.reversed ? ' (Reversed)' : '')
    );
  }).join('\n');

  var userQuestion = payload.userQuestion
    ? 'The querent\'s specific question: ' + payload.userQuestion
    : 'The querent did not provide a specific written question.';

  return [
    'You are a tarot reader channeling the deep wisdom of Rachel Pollack (Seventy-Eight Degrees of Wisdom), Stefanie Caponi (Guided Tarot), and Liz Dean (The Ultimate Guide to Tarot).',
    'You know every card\'s symbolism, traditional meaning, reversed nuances, and spread dynamics inside out.',
    '',
    'The querent\'s topic: ' + payload.category,
    userQuestion,
    'Spread used: ' + payload.formation.name,
    '',
    'Cards drawn (in position order):',
    cardLines,
    '',
    '---',
    '',
    'Deliver this tarot reading EXCLUSIVELY in Reddit roast mode. Follow these rules precisely:',
    '',
    '1. Open with an ALL-CAPS Reddit-style post title on its own line.',
    'Example: "PULLING THE TOWER CARD THREE TIMES THIS WEEK?? (not clickbait) or a simular opening thats puchy"',
    '',
    '2. Start the reading body with: "So I pulled these cards for you and I am not okay."or a simular opening thats puchy based on the cards that is drawn',
    '',
    '3. For EACH card in position order, write a section with this exact header format: **[Position Name]: [Card Name]**. Add " (Reversed)" after the card name if applicable.',
    'Under each header, write 2–3 sentences of genuine interpretation, then add one sarcastic Reddit-style observation.',
    '',
    'Use Reddit-isms naturally: r/YourLife, big yikes, RemindMe! 6 months, this is the way, edit: wow this blew up, certified YTA behavior.',
    '',
    '4. End with a **Overall Verdict:** section that synthesizes all cards in context of the querent\'s topic.',
    '',
    '5. Close with a optimistically pessimistic Reddit edit in italics. you live in 2026 but dont use phases that people NOT online wouldnt understand',
    '',
    '6. Do not break character. No "As an AI..." disclaimers. No apologies. You are the Oracle of Reddit.',
    '',
    '7. Total length: 450–700 words. Make it spicy, but make the tarot wisdom genuinely shine through the roasting.'
  ].join('\n');
}

// ── HTML FORMATTER ────────────────────────────────────────────────────────────

function formatReadingAsHtml(text) {
  var html = text
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;');

  html = html.replace(/\*\*([^*\n]+)\*\*/g, '<strong>$1</strong>');
  html = html.replace(/\*([^*\n]+)\*/g, '<em>$1</em>');

  var blocks = html.split(/\n\n+/);

  var parts = blocks.map(function(block) {
    block = block.trim();

    if (!block) {
      return '';
    }

    if (/^<strong>[^<]+<\/strong>$/.test(block)) {
      return '<h3 class="reading-section-header">' + block + '</h3>';
    }

    block = block.replace(/\n/g, '<br>');
    return '<p>' + block + '</p>';
  });

  return parts.filter(function(p) {
    return p;
  }).join('\n');
}
